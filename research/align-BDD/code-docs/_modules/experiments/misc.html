import sys


[docs]def log(instance, *what_to_print, outfile = sys.stdout, comment="--none--"):
    outfile.write(str(instance))
    for val in list(what_to_print):
        outfile.write(",")
        outfile.write(str(val))

    outfile.write(",{}\n".format(comment))
    if outfile == sys.stdout:
        outfile.flush()
